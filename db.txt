CREATE TABLE SENDERKEY_TB (
 MASTERID varchar2(16) not null,
 SENDERKEY varchar2(40) not null,
 SENDERKEY_TYPE char(1) default 'S' not null,
 PROFILE_UUID varchar2(50) not null,
 PROFILE_NAME varchar2(60) not null,
 REG_DATE date not null,
 MOD_DATE date null,
 REG_ID varchar2(16) not null,
 MOD_ID varchar2(16) null,
 USE_YN char(1) default 'Y' not null,
 CONSTRAINT SENDERKEY_TB_IDX01 PRIMARY KEY ("MASTERID", "SENDERKEY")
);

CREATE TABLE TEMPLATE_DATA (
 TEMPLATE_SEQ number(11,0) not null,
 SENDERKEY varchar2(40) not null,
 TEMPLATE_CODE varchar2(30) not null,
 TEMPLATE_NAME varchar2(200) not null,
 TEMPLATE_CONTENT varchar2(2000) not null,
 INSPECTION_STATUS char(3),
 CREATED_AT date null,
 MODIFIED_AT date null,
 REG_DATE date default sysdate,
 MOD_DATE date null,
 REG_ID varchar2(16) not null,
 MOD_ID varchar2(16) null,
 STATUS char(1),
 CUR_STATUS number(2,0),
 PROCESS_ING char(1) default 'N',
 CONSTRAINT TEMPLATE_DATA_IDX01 PRIMARY KEY ("TEMPLATE_SEQ")
);

CREATE TABLE TEMPLATE_COMMENTS (
 COMMENTS_SEQ number(11,0) not null,
 TEMPLATE_CODE varchar2(30) not null,
 ID number(11,0),
 CONTENT varchar2(2000),
 USER_NAME varchar2(100),
 CREATED_AT date,
 STATUS char(3),
 CONSTRAINT TEMPLATE_COMMENTS_IDX01 PRIMARY KEY ("COMMENTS_SEQ")
);

CREATE TABLE TEMPLATE_BUTTONS (
 TEMPLATE_CODE varchar2(30) not null,
 ORDERING number(1,0) not null,
 BUTTON_NAME varchar2(28) not null,
 LINKTYPE char(2) not null,
 LINKMO varchar2(500),
 LINKPC varchar2(500),
 LINKIOS varchar2(500),
 LINKAND varchar2(500),
 CONSTRAINT TEMPLATE_BUTTONS_IDX01 PRIMARY KEY ("TEMPLATE_CODE", "ORDERING")
);

CREATE TABLE TEMPLATE_BUTTONS_TYPE (
 LINKTYPE char(2) not null,
 LINKTYPENAME varchar2(20) not null
);

CREATE TABLE TEMPLATE_HISTORY (
 TEMPLATE_CODE varchar2(30) not null,
 CUR_STATUS number(2,0),
 INSPECTION_STATUS char(3),
 REG_DATE date,
 RESULT_CODE number(5,0),
 RESULT_DESC varchar2(500)
);

CREATE SEQUENCE TEMPLATE_DATA_SEQ01 MINVALUE 1 MAXVALUE 99999999999 INCREMENT BY 1 NOCACHE NOORDER NOCYCLE;
CREATE SEQUENCE TEMPLATE_COMMENTS_SEQ01 MINVALUE 1 MAXVALUE 99999999999 INCREMENT BY 1 NOCACHE NOORDER NOCYCLE;
CREATE INDEX TEMPLATE_HISTORY_IDX01 ON TEMPLATE_HISTORY (TEMPLATE_CODE);
